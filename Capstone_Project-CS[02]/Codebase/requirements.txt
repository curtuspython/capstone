# ---- Structured resume extraction (two-tier approach) -----------------------
#
# Tier 1: pyresparser (project requirement — attempted first on every CV)
#   NOTE: pyresparser targets spaCy 2.x and ships a config.cfg that is
#   incompatible with spaCy 3.x, raising [E053]. resume_parser.py catches
#   this error and falls through to Tier 2 automatically.
pyresparser>=1.0.6
nltk>=3.8.0               # required by pyresparser
#
# Tier 2: spaCy 3.x NER + regex (fallback — always works on Python 3.11+)
spacy>=3.0.0
# After install: python -m spacy download en_core_web_sm
# (resume_parser.py auto-downloads this model on first run)

# ---- Semantic similarity fallback -------------------------------------------
# Used when Gemini text-embedding-004 is unavailable (e.g. API quota/version).
# TF-IDF cosine similarity via scikit-learn produces real per-candidate scores.
scikit-learn>=1.3.0

# ---- PDF report generation --------------------------------------------------
fpdf2>=2.8.0
